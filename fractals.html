The Koch Curve

This is the most basic form of the Koch curve:

def koch_curve(t, depth, size):
    if depth == 1:
        t.fd(size)
    else:
        t.color("blue")
        koch_curve(t, depth-1, size / 1.5)
        t.lt(60)
        t.color("orange")
        koch_curve(t, depth-1, size / 0.75)
        t.rt(120)
        t.color("green")
        koch_curve(t, depth-1, size / 0.75)
        t.lt(60)
        t.color("magenta")
        koch_curve(t, depth-1, size / 1.5)

The Sierpinski Triangle

The most basic version of the Sierpinski Triangle:

def sierpinski(t, depth, size):

        triangle(t, size)

The Fractal Tree

The most basic Fractal Tree:

def tree(t, depth, size, angle):
    if depth == 0:
        t.color("royalblue")
        t.fd(size)
        t.bk(size)
        t.color("orangered")
        t.fd(size)
        t.bk(size)
        t.color("royalblue")
        t.fd(size)
        t.bk(size)
        t.color("orangered")
    else:
        t.color("royalblue")
        t.fd(size)
        t.rt(angle)
        tree(t, depth-1, size, angle)
        t.color("orangered")
        t.lt(2 * angle)
        tree(t, depth-1, size, angle)
        t.rt(angle)
        t.bk(size)
        t.rt(180)
        t.color("springgreen")
        t.fd(size)
        t.rt(angle)
        tree(t, depth-1, size, angle)
        t.color("purple")
        t.lt(2 * angle)
        tree(t, depth-1, size, angle)
        t.rt(angle)
        t.bk(size)
